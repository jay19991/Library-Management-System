CREATE TABLE L_STUDENT
(STUDENTID VARCHAR(30) CONSTRAINT L_STUDENT_STUDENTID_PK PRIMARY KEY,
FNAME VARCHAR(30) CONSTRAINT L_STUDENT_FNAME_NN NOT NULL,
LNAME VARCHAR(30) CONSTRAINT L_STUDENT_LNAME_NN NOT NULL,
BIRTHDATE DATE,
STATE CHAR(2) CONSTRAINT L_STUDENT_STATE_CK CHECK (STATE IN('NY','NJ','CT','CA')),
ZIPCODE NUMBER(5) CONSTRAINT L_STUDENT_ZIPCODE_NN NOT NULL,
EMAILID VARCHAR(30) CONSTRAINT L_STUDENT_EMAILID_NN NOT NULL,
MOBILENUM NUMBER(10) CONSTRAINT L_STUDENT_MOBILENUM_NN NOT NULL,
DEPTID VARCHAR(30) CONSTRAINT L_STUDENT_DEPTID_NN NOT NULL,
DEPTNAME VARCHAR(30) CONSTRAINT L_STUDENT_DEPTNAME_NN NOT NULL,
CURRENTYEAR VARCHAR(30) CONSTRAINT L_STUDENT_CURRENTYEAR_NN NOT NULL
);

CREATE TABLE L_STAFF
(STAFFID VARCHAR(30) CONSTRAINT L_STAFF_STAFFID_PK PRIMARY KEY,
FNAME VARCHAR(30) CONSTRAINT L_STAFF_FNAME_NN NOT NULL,
LNAME VARCHAR(30) CONSTRAINT L_STAFF_LNAME_NN NOT NULL,
DEPTID VARCHAR(30) CONSTRAINT L_STAFF_DEPTID_UK UNIQUE,
DEPTNAME VARCHAR(30) CONSTRAINT L_STAFF_DEPTNAME_NN NOT NULL,
STATE CHAR(2) CONSTRAINT L_STAFF_STATE_CK CHECK (STATE IN('NY','NJ','CT','CA')),
ZIPCODE NUMBER(5) CONSTRAINT L_STAFF_ZIPCODE_NN NOT NULL,
EMAILID VARCHAR(30) CONSTRAINT L_STAFF_EMAILID_NN NOT NULL,
MOBILENUM NUMBER(10) CONSTRAINT L_STAFF_MOBILENUM_NN NOT NULL
);

CREATE TABLE L_BOOK
(BOOKNUM VARCHAR(30) CONSTRAINT L_BOOK_BOOKNUM_PK PRIMARY KEY,
BOOKTITLE VARCHAR(30) CONSTRAINT L_BOOK_BOOKTITLE_NN NOT NULL,
ISBNNUM VARCHAR(30) CONSTRAINT L_BOOK_ISBNNUMBER_UK UNIQUE,
AUTHORNAME VARCHAR(30) CONSTRAINT L_BOOK_AUTHORNAME_NN NOT NULL,
PUBNAME VARCHAR(30) CONSTRAINT L_BOOK_PUBLICATION_NN NOT NULL,
PUBYEAR VARCHAR(30) CONSTRAINT L_BOOK_PUBYEAR_NN NOT NULL,
EDITION VARCHAR(30) CONSTRAINT L_BOOK_EDITION_NN NOT NULL,
DEPTBOOK VARCHAR(30) CONSTRAINT L_BOOK_DEPTBOOK_NN NOT NULL,
COPIESLEFT CHAR(1) CONSTRAINT L_BOOK_COPIESLEFT_CK CHECK (COPIESLEFT IN ('Y','N'))
);

CREATE TABLE L_ISSUEDETAILS
(ISSUENUM VARCHAR(30) CONSTRAINT L_ISSUEDETAILS_ISSUENAME_PK PRIMARY KEY,
ISSUEDATE DATE CONSTRAINT L_ISSUEDETAILS_ISSUEDATE_NN NOT NULL,
BOOKNUM VARCHAR(30),
DUEDATE DATE CONSTRAINT L_ISSUEDETAILS_DUEDATE_NN NOT NULL,
STUDENTID VARCHAR(30),
STAFFID VARCHAR(30),
DEPTID VARCHAR(30) 
);

CREATE TABLE L_DEPARTMENT
(DEPTID VARCHAR(30) CONSTRAINT L_DEPARTMENT_DEPTID_PK PRIMARY KEY,
DEPTNAME VARCHAR(30) CONSTRAINT L_DEPARTMENT_DEPTNAME_NN NOT NULL,
STUDENTID VARCHAR(30),
STAFFID VARCHAR(30)
);
